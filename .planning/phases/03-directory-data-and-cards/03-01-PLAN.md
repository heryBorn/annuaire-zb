import { useState, useEffect } from 'react';

export function useMemberFetch() {
  const [members, setMembers] = useState([]);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState(null);

  useEffect(() => {
    const controller = new AbortController();
    const url = process.env.REACT_APP_SHEET_API_URL + '?action=getMembers';

    fetch(url, { signal: controller.signal })
      .then(res => res.json())
      .then(data => {
        setMembers(data.members || []);
        setLoading(false);
      })
      .catch(err => {
        if (err.name === 'AbortError') return;
        setError(err.message);
        setLoading(false);
      });

    return () => controller.abort();
  }, []);

  return { members, loading, error };
}
```

Critical rules:
- Named export `useMemberFetch` (not default export) — consumers will import { useMemberFetch }
- Always use `data.members || []` fallback — GAS response shape unvalidated until live test
- AbortController cleanup is mandatory — React 18 StrictMode double-invokes effects
- Do NOT use `mode: 'no-cors'` — the existing index.html proves a plain GET works and no-cors produces opaque responses that block res.json()
- URL is REACT_APP_SHEET_API_URL (already correct in annuaire-zb-react/.env) — never hardcode the URL

function getAvailStyle(d) {
  if (!d) return 'grey';
  if (d.includes('Disponible') && !d.includes('Partiel')) return 'green';
  if (d.includes('Partiel') || d.includes('Recherche')) return 'orange';
  return 'grey';
}

const badgeConfig = {
  green:  { dot: 'bg-sage',  bg: 'bg-sage/20',  label: 'Disponible'        },
  orange: { dot: 'bg-wheat', bg: 'bg-wheat/20', label: 'Partiellement'      },
  grey:   { dot: 'bg-muted', bg: 'bg-muted/20', label: 'Non disponible'     },
};

function AvailabilityBadge({ disponibilite }) {
  const key = getAvailStyle(disponibilite);
  const { dot, bg, label } = badgeConfig[key];

  return (
    <span className={`inline-flex items-center gap-1.5 px-2 py-0.5 rounded-full text-xs font-sans font-medium ${bg} mt-2`}>
      <span className={`w-2 h-2 rounded-full flex-shrink-0 ${dot}`} />
      {label}
    </span>
  );
}

export default AvailabilityBadge;
```

Critical rules:
- `getAvailStyle` logic must match exactly the existing index.html `availClass()` function — this is the authoritative business rule
- Use project Tailwind color tokens: `bg-sage`, `bg-wheat`, `bg-muted` (not raw Tailwind colors like green-500)
- Use `badgeConfig` lookup object — do NOT inline ternaries in JSX className
- Default export (component), since it will always be used as `import AvailabilityBadge from '...'`
- This component will be reused in Phase 4 modal — keep it purely presentational, no phase-specific coupling
